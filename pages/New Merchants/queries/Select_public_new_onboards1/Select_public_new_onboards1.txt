SELECT
  *
FROM
  public.new_onboards
WHERE
  "phone" ilike '%{{fetch_new_merchants.searchText}}%'
ORDER BY
  "{{fetch_new_merchants.sortOrder.column || 'id'}}" {{fetch_new_merchants.sortOrder.order !== "desc" ? "" : "DESC"}}
LIMIT
  {{fetch_new_merchants.pageSize}}
OFFSET
  {{fetch_new_merchants.pageOffset}}
